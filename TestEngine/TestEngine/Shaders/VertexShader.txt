#version 330 core

layout (location = 0) in vec2 inPosition;
layout (location = 1) in vec2 inTexCoord;
layout (location = 2) in vec4 inColor;

out vec2 TexCoord;
out vec4 Color;

uniform vec2 uv;

uniform mat4 world;
uniform mat4 projection;
uniform mat4 view;

void main(void) 
{ 
    gl_Position = projection * world * vec4(inPosition, 0, 1);
    TexCoord = uv + inTexCoord;
    Color = inColor;
}
